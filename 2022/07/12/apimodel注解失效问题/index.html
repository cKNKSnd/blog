Swagger @ApiModel 注解失效问题

Kendall Cai   in Amoy
2022-07-12
  • Springboot
  • 问题记录

# 问题描述

配置了Swagger的Springboot项目中，@ApiModel注解失效，在swagger-ui.html的网页文档中无法查看到配置了该注解的实体类。

# 解决方案

Restful API 请求中，返回的包装类需要使用泛型。然后就可以在swagger-ui.html中查看到对应的实体类信息。

# 包装类泛型示例


/**
 * Result.java
 * @Description Restful API Request Wrapper class
 */
@Data
public class Result<T> {

    /**
     * 响应状态码
     */
    private String status;

    /**
     * 响应信息
     */
    private String msg;

    /**
     * 响应结构
     */
    private T data;

    public Result() {

    }

    public Result(IBaseErrorInfo errorInfo) {
        this.status = errorInfo.getResultStatus();
        this.msg = errorInfo.getResultMsg();
    }

    /**
     * 成功
     *
     * @return
     */
    public static Result<?> success() {
        return success(null);
    }

    /**
     * 成功
     *
     * @param data
     * @return
     */
    public static  <T> Result<T> success(T data) {
        Result<T> result = new Result<>();
        result.setStatus(CommonEnum.SUCCESS.getResultStatus());
        result.setMsg(CommonEnum.SUCCESS.getResultMsg());
        result.setData(data);
        return result;
    }

    /**
     * 失败
     *
     * @param status
     * @param msg
     * @return
     */
    public static Result<?> error(String status, String msg) {
        Result<?> result = new Result<>();
        result.setStatus(status);
        result.setMsg(msg);
        result.setData(null);
        return result;
    }

    /**
     * 失败
     *
     * @param errorInfo
     * @param <?>
     * @return
     */
    public static Result<?> error(IBaseErrorInfo errorInfo) {
        Result<?> result = new Result<>();
        result.setStatus(errorInfo.getResultStatus());
        result.setMsg(errorInfo.getResultMsg());
        result.setData(null);
        return result;
    }

    /**
     * 失败
     *
     * @param msg
     * @return
     */
    public static Result<?> error(String msg) {
        Result<?> result = new Result<>();
        result.setStatus("-1");
        result.setMsg(msg);
        result.setData(null);
        return result;
    }
}



/**
 * IBaseErrorInfo.java
 * @Description 基础错误接口
 */
public interface IBaseErrorInfo {
    /**
     * 获取错误码
     * @return
     */
    String getResultStatus();

    /**
     * 获取错误描述
     * @return
     */
    String getResultMsg();
}

package common.result;

/**
 * CommonEnum.java
 * @Description Common HTTP Response Status
 */
public enum CommonEnum implements IBaseErrorInfo {
    // 数据操作错误定义
    SUCCESS("0", "(copdsp)Success!"),
    BODY_NOT_MATCH("400", "请求的数据格式不符！"),
    UNAUTHORIZED("401", "未授权！"),
    NOT_FOUND("404", "未找到该资源！"),
    INTERNAL_SERVER_ERROR("500", "服务器内部错误！")
    ;

    /**
     * 错误码
     */
    private String resultStatus;

    /**
     * 错误描述
     */
    private String resultMsg;

    CommonEnum(String resultStatus, String resultMsg) {
        this.resultStatus = resultStatus;
        this.resultMsg = resultMsg;
    }

    /**
     * 获取错误码
     *
     * @return
     */
    @Override
    public String getResultStatus() {
        return this.resultStatus;
    }

    /**
     * 获取错误描述
     *
     * @return
     */
    @Override
    public String getResultMsg() {
        return this.resultMsg;
    }
}
